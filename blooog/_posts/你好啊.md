function debounce(func, delay = 300, immediate = true) {
  let timer = 0;
  return function () {
    immediate && !timer && func.apply(this, arguments);
    clearTimeout(timer);
    timer = setTimeout(() => {
      func.apply(this, arguments);
    }, delay);
  };
}
